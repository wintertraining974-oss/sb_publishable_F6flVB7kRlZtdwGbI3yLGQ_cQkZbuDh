function dk(d) {
  return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
}
function parseDate(k) {
  const [y,m,d] = k.split('-').map(Number);
  return new Date(y, m-1, d);
}
function isClosed(d) {
  const k = dk(d);
  if (d.getDay() === 0) return true;  // Dimanche toujours fermé
  if (HOLIDAYS[k]) return true;        // Jour férié à sa date exacte
  if (S.customClosed[k]) return true;
  return false;
}
function outOfRange(d) { return d < START || d > END; }
function holidayName(d) {
  const k = dk(d);
  if (d.getDay() === 0) return 'Dimanche';
  return HOLIDAYS[k] || (S.customClosed[k] ? 'Fermé (admin)' : '');
}
function allSlots() { return [...S.morningOrder, ...S.afternoonOrder]; }
function getSlot(dateK, slot) {
  if (!S.data[dateK]) S.data[dateK] = {};
  if (!S.data[dateK][slot]) S.data[dateK][slot] = [];
  return S.data[dateK][slot];
}
function cap(slot) { return S.capacity[slot] || 15; }

function isLocked(dateK, slot) {
  if (!S.priorityOn || S.isAdmin) return false;
  const groups = [S.morningOrder, S.afternoonOrder];
  for (const g of groups) {
    const idx = g.indexOf(slot);
    if (idx < 0) continue;
    for (let i = 0; i < idx; i++) {
      if (getSlot(dateK, g[i]).length < cap(g[i])) return true;
    }
    return false;
  }
  return false;
}

function slotStatus(dateK, slot) {
  if (isLocked(dateK, slot)) return 'locked';
  const n = getSlot(dateK, slot).length;
  const c = cap(slot);
  if (n >= c) return 'full';
  if (n / c >= 0.6) return 'mid';
  return 'open';
}

function dayStatus(d) {
  if (isClosed(d) || outOfRange(d)) return null;
  const k = dk(d);
  let total = 0, totalCap = 0;
  allSlots().forEach(s => {
    total += getSlot(k, s).length;
    totalCap += cap(s);
  });
  if (total === 0) return 'green';
  if (total / totalCap >= 0.65) return 'red';
  return 'orange';
}

function weekStart(d) {
  const dt = new Date(d);
  const day = dt.getDay();
  dt.setDate(dt.getDate() - (day === 0 ? 6 : day - 1));
  return dt;
}
function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function renderMonth() {
  const d = S.currentDate;
  document.getElementById('calTitle').textContent = `${MONTHS_FR[d.getMonth()]} ${d.getFullYear()}`;
  const first = new Date(d.getFullYear(), d.getMonth(), 1);
  const offset = (first.getDay() + 6) % 7;
  const daysCount = new Date(d.getFullYear(), d.getMonth()+1, 0).getDate();
  let h = '<div class="month-grid">';
  ['Lun','Mar','Mer','Jeu','Ven','Sam','Dim'].forEach(x => h += `<div class="cal-head">${x}</div>`);
  for (let i = 0; i < offset; i++) h += '<div class="cal-cell cc-empty"></div>';
  for (let n = 1; n <= daysCount; n++) {
    const dt = new Date(d.getFullYear(), d.getMonth(), n);
    const k = dk(dt);
    const isHol = !!HOLIDAYS[k];
    const closed = isClosed(dt);
    const oor = outOfRange(dt);
    const sel = S.selectedDate && dk(S.selectedDate) === k;
    const st = dayStatus(dt);
    const label = holidayName(dt);
    let cls = 'cal-cell';
    if (isHol) cls += ' cc-holiday';
    else if (closed || oor) cls += ' cc-closed';
    if (sel && !closed && !oor) cls += ' cc-selected';
    const click = (!closed && !oor) ? `onclick="selectDay('${k}')"` : '';
    h += `<div class="${cls}" ${click}>
      <div class="cc-num">${n}</div>
      ${label ? `<div class="cc-tag">${esc(label)}</div>` : ''}
      ${!closed && !oor && st ? `<div class="cc-dot d-${st}"></div>` : ''}
    </div>`;
  }
  h += '</div>';
  return h;
}

function renderWeek() {

  document.getElementById('calTitle').textContent =

  let h = '<div class="week-grid">';

    const dt = new Date(ws); dt.setDate(ws.getDate()+i);

    const closed = isClosed(dt) || outOfRange(dt);
    const sel = S.selectedDate && dk(S.selectedDate) === k;
    const isHol = !!HOLIDAYS[k];
    let cls = 'wk-col' + (sel ? ' wk-selected' : '') + (closed ? ' wk-closed' : '');

        <div class="wk-daynum" style="${isHol ? 'color:var(--red)' : ''}">${dt.getDate()}</div>

    if (closed) {

          <div class="wk-slot-time">${slot}</div>

        </div>`;
      });
      h += `<div class="wk-section" style="border-top:1px solid var(--border)">Après-midi</div>`;
      S.afternoonOrder.forEach(slot => {
        const aths = getSlot(k, slot);
        const c = cap(slot);
        const pct = Math.min(1, aths.length / c);
        const locked = isLocked(k, slot) && !S.isAdmin;
        const color = pct >= 1 ? 'var(--red)' : pct >= .6 ? 'var(--orange)' : 'var(--green)';
        h += `<div class="wk-slot${locked ? ' wk-locked' : ''}">
          <div class="wk-slot-time">${slot}</div>
          <div class="wk-slot-count">${aths.length}/${c}</div>
          <div class="wk-bar" style="background:var(--border2)"><div style="width:${pct*100}%;height:100%;background:${color};border-radius:2px"></div></div>

function renderSlot(dateK, slot) {
  const aths = getSlot(dateK, slot);
  const c = cap(slot);
  const st = slotStatus(dateK, slot);
  const pct = Math.min(1, aths.length / c);
  const exp = S.expanded[`${dateK}-${slot}`];
  const labels = { open:'Disponible', mid:'Presque plein', full:'Complet', locked:'🔒 Verrouillé' };
  const bCls   = { open:'sb-open', mid:'sb-mid', full:'sb-full', locked:'sb-locked' };

  // Find prev slot for locked msg
  let prevSlot = '';
  if (st === 'locked') {
    const g = S.morningOrder.includes(slot) ? S.morningOrder : S.afternoonOrder;
    const idx = g.indexOf(slot);
    if (idx > 0) prevSlot = g[idx-1];
  }

  let h = `<div class="slot-card s-${st}">
    <div class="slot-header" onclick="toggleSlot('${dateK}','${slot}')">
      <div class="slot-time">${slot}</div>
      <div class="slot-count-text">${aths.length} / ${c} inscrits</div>
      <div class="slot-progress"><div class="slot-progress-fill" style="width:${pct*100}%"></div></div>
      <span class="slot-badge ${bCls[st]}">${labels[st]}</span>
      <span class="slot-chevron${exp ? ' open' : ''}">▼</span>
    </div>`;

  if (st === 'locked' && !S.isAdmin) {
    h += `<div class="locked-msg">⏳ S'ouvre après remplissage du créneau ${prevSlot}</div>`;
  }

  if (exp) {
    h += `<div class="slot-body">
      <div class="athletes-list">`;
    if (aths.length === 0) {
      h += `<div class="empty-slot">Aucun inscrit pour le moment.</div>`;
    } else {
      aths.forEach((a, i) => {
        h += `<div class="ath-row">
          <span class="ath-num">${i+1}</span>
          <span class="ath-name">${esc(a.name)}</span>
          <span class="ath-role-badge ${a.role === 'Encadrant' ? 'role-encadrant' : 'role-athlete'}">${esc(a.role)}</span>
          <div class="ath-actions">
            ${S.isAdmin ? `<button class="ath-btn mv" title="Déplacer" onclick="openMove('${dateK}','${slot}',${i})">↔</button>` : ''}
            <button class="ath-btn del" title="Supprimer" onclick="removeAthlete('${dateK}','${slot}',${i})">✕</button>
          </div>
        </div>`;
      });
    }
    h += `</div>`;

    if (st !== 'full' && !(st === 'locked' && !S.isAdmin)) {
      h += `<div class="add-form">
        <input class="field" type="text" id="n-${dateK}-${slot}" placeholder="Nom Prénom">
        <select class="field" id="r-${dateK}-${slot}">
          <option value="Athlète">Athlète</option>
          <option value="Encadrant">Encadrant</option>
        </select>
        <button class="btn btn-sm btn-primary" onclick="addAthlete('${dateK}','${slot}')">S'inscrire</button>
      </div>`;
    } else if (st === 'full') {
      h += `<div class="full-msg">Créneau complet — plus de places disponibles.</div>`;
    }
    h += `</div>`;
  }
  h += `</div>`;
  return h;
}

function toggleSlot(dateK, slot) {
  const key = `${dateK}-${slot}`;

}

function addAthlete(dateK, slot) {
  const nameEl = document.getElementById(`n-${dateK}-${slot}`);
  const roleEl = document.getElementById(`r-${dateK}-${slot}`);
  if (!nameEl) return;
  const name = nameEl.value.trim();
  if (!name) { toast('Entrez un nom et prénom.', 'error'); return; }
  const c = cap(slot);
  if (getSlot(dateK, slot).length >= c) { toast('Créneau complet.', 'error'); return; }
  getSlot(dateK, slot).push({ name, role: roleEl.value });
  save();
  toast(`${name} inscrit(e) sur ${slot} !`, 'success');
  renderDayPanel();
  renderCal();
}

function removeAthlete(dateK, slot, idx) {
  const name = getSlot(dateK, slot)[idx].name;
  getSlot(dateK, slot).splice(idx, 1);
  save();
  toast(`${name} désinscrit(e).`, 'success');
  renderDayPanel();
  renderCal();
}

function movePrio(group, idx, dir) {
  const arr = group === 'morning' ? S.morningOrder : S.afternoonOrder;
  const ni = idx + dir;
  if (ni < 0 || ni >= arr.length) return;
  [arr[idx], arr[ni]] = [arr[ni], arr[idx]];
  save();
  render();
}

  S.capacity[slot] = Math.max(1, parseInt(val) || 15);
  save();
  renderDayPanel();
}

  render();
}

function confirmMove() {
  if (!S.movePending) return;
  const { dateK, slot, idx, name, role } = S.movePending;
  const nd = document.getElementById('moveDate').value;
  const ns = document.getElementById('moveSlot').value;
  if (!nd) { toast('Choisissez une date cible.', 'error'); return; }
  if (getSlot(nd, ns).length >= cap(ns)) { toast('Créneau cible complet.', 'error'); return; }
  getSlot(dateK, slot).splice(idx, 1);
  getSlot(nd, ns).push({ name, role });

  closeModal('moveModal');
  S.movePending = null;
  render();
}

  const k = dk(d);
  const dayName = DAYS_FULL[d.getDay()];
  let t = `📅 Planning Athlétisme\n${dayName} ${d.getDate()} ${MONTHS_FR[d.getMonth()]} ${d.getFullYear()}\n${'─'.repeat(30)}\n\n`;
  const slots = allSlots();
  slots.forEach(slot => {
    const aths = getSlot(k, slot);
    const c = cap(slot);
    const st = slotStatus(k, slot);
    if (st === 'locked') return;
    const icon = st === 'full' ? '🔴' : st === 'mid' ? '🟡' : '🟢';
    t += `${icon} ${slot} — ${aths.length}/${c} inscrits\n`;
    aths.forEach((a, i) => t += `   ${i+1}. ${a.name} (${a.role})\n`);
    if (aths.length === 0) t += `   (aucun inscrit)\n`;
    t += '\n';
  });
  return t.trim();
}

function copyText() {
  const t = buildText();
  if (!t) return;
  navigator.clipboard.writeText(t)
    .then(() => toast('Copié dans le presse-papiers !', 'success'))
    .catch(() => toast('Erreur de copie.', 'error'));
}

function sendWA() {
  const t = buildText();
  if (!t) return;
  window.open('https://wa.me/?text=' + encodeURIComponent(t), '_blank');
}

function toast(msg, type='') {
  const el = document.getElementById('toast');
  el.textContent = msg;
  el.className = 'toast show' + (type ? ` t-${type}` : '');
  clearTimeout(el._t);
  el._t = setTimeout(() => el.className = 'toast', 2800);
}
